exercice 1 :
procédure saisie(@taille_T : entier, @taille_Tm : entier)
  début
    répeter
     écrire("entrez la taille du tableau T :")
     lire(taille_T)
    jusqu'à (taille_T > 2 et taille_T <= 20)

    répeter
     écrire("entrez	la taille du tableau Tm : ")
     lire(taille_Tm)
    jusqu'à (taille_Tm > 2 et taille_Tm <= 200)
   fin

fonction Comparaison (ch : chaine , ch_cherchée : chaine) : booléen
   variables : 
      i, j: entier
      trouvée : booléen
   début 
      trouvé <-- faux
      pour i de 0 de long(ch) - 1 faire
       pour j de 0 de long(ch_cherchée) - 1 faire
         si (ch[i] = ch_cherchée[j]) alors
           si (ch[i+1] = ch_cherchée[j+1]) alors
             si (ch[i+2] = ch_cherchée[j+2]) alors
               trouvée <-- vrai
             sinon 
               trouvée <-- faux
             finSi
           sinon
             trouvée <-- faux
           finSi
         sinon
           trouvée <--faux
         finSi
       finPour
      finPour
      retourner trouvée
   fin
