exercice n2 :
algorithm Moyenne
    var:
      matiére1 , matiére2 , matiére3 : réels
      coefficient_matiére1 , coefficient_matiére2 , coefficient_matiére3 : réels
début

    écrire("Entrer la note matiére1 :")
    lire(matiére1)
    écrire("Entrer la note matiére2 :")
    lire(matiére2)
    écrire("Entrer la note matiére3 :")
    lire(matiére3)
    
   
    écrire("Entrer la note coefficientde la matiére1 :")
    lire(coefficient_matiére1)
    écrire("Entrer la note coefficient de la matiére2 :")
    lire(coefficient_matiére2)
    écrire("Entrer la note coefficient de la matiére3 :")
    lire(coefficient_matiére3)
       
    somme1 <--(matiére1*coefficient_matiére1 + matiére2*coefficient_matiére2 + matiére3*coefficient_matiére3)
    somme2 <--(coefficient_matiére1 + coefficient_matiére2 + coefficient_matiére3)
    moyenne <-- somme1/somme2  
    écrire("la moyenne est :", moyenne)
    
fin

exercice n4:

 différence <-- a-b

 différence <-- b-a

 b <--b-différence
