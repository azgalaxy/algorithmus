		Procédure Saisir_taille(var nombre : entier)
			Début
				Répéter
					Ecire("Donnez la valeur de nombre : ")
					Lire(nombre)
				Jusqu'à(nombre > 0 ou nombre < 3)
			Fin Procédure
		
		Procédure Saisir_matrice(var matrice[3][3] : , nombre : entier)
			Variable :
				i, j : entier
			
			Début
				Pour i de 1 à nombre Faire
					Pour j de 1 à nombre Faire
						Ecrire("matrice[",i,"][",j,"] = ")
						Lire(matrice[i][j])
					Fin Pour
				Fin Pour
			Fin Procédure
			
		Procédure Affichage(matrice[3][3] : , nombre : entier)
			Variable :
				i, j : entier
			
			Début
				Pour i de 1 à nombre Faire
					Pour j de 1 à nombre Faire
						Ecrire(matrice[i][j]," | ")
					Fin Pour
				Fin Pour
			Fin Procédure
			
		Procédure Produit(matrice1[3][3] : , matrice2[3][3] : , nombre : entier)
			Variable : 
				i, j, k :entier
				sum[3][3] : matrice d'entiers
				somme : entier
				
			Début
				somme <-- 0
				Pour i de 1 à nombre Faire
					Pour j de 1 à nombre Faire
						Pour k de 1 à nombre Faire
							somme += matrice1[i][k] * matrice2[k][j]
						Fin Pour
						sum[i][j] <-- somme
					Fin Pour
				Fin Pour
				Fin Procédure
					
			
		Algorithme Principale
			Variable :
				taille : entier
			Début
				Saisir_taille(taille)
