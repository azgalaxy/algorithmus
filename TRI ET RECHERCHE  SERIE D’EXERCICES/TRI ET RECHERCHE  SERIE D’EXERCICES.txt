    Afficher "La taille du tableau doit être entre 4 et 31."
    Arrêter
  Fin Si



  Pour i allant de 0 à N-1 Faire
    Pour j allant de i+1 à N-1 Faire
      somme_i <- 0
      somme_j <- 0
      


      nombre_i <- T[i]
      Tant que nombre_i > 0 Faire
        somme_i <- somme_i + (nombre_i % 10)
        nombre_i <- nombre_i / 10
      Fin Tant que
     

      nombre_j <- T[j]
      Tant que nombre_j > 0 Faire
        somme_j <- somme_j + (nombre_j % 10)
        nombre_j <- nombre_j / 10
      Fin Tant que
   

      Si somme_i < somme_j Alors
        temp <- T[i]
        T[i] <- T[j]
        T[j] <- temp
      Fin Si
    Fin Pour
  Fin Pour

  Afficher "Tableau trié dans l'ordre décroissant en fonction de la somme des chiffres :"
  Pour i allant de 0 à N-1 Faire
