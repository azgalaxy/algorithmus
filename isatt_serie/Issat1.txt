// EXERCICE1
Algorithme moyen
Variables
    n1: réel
    n2: réel
    n3: réel
    moyen: réel
    np1: bool
    np2: bool
    np3: bool
Début
    Ecrire("Entrez la première note : ")
    Lire(n1)

    Ecrire("Entrez la deuxième note : ")
    Lire(n2)

    Ecrire("Entrez la troisième note : ")
    Lire(n3)

    Si n1 > 9 Alors
        np1 <- vrai
    Sinon
        np1 <- faux
    FinSi

    Si n2 > 9 Alors
        np2 <- vrai
    Sinon
        np2 <- faux
    FinSi

    Si n3 > 9 Alors
        np3 <- vrai
    Sinon
        np3 <- faux
    FinSi

    Si np1 = vrai et np2 = vrai et np3 = vrai Alors
        Ecrire("Admis")
    FinSi

    moyen <- n1 + n2 + n3

    Si n1 = 8 ou n2 = 8 ou n3 = 8 Alors
        Si moyen >= 10 Alors
            Ecrire("Admis")
        FinSi
    FinSi
Fin

// EXERCICE2
Algorithme salary
Variables
    su: réel
    sd: réel
    s: réel
    s1: réel
    s2: réel
    s3: réel
Début
    Ecrire("Le prix unitaire d'une heure : ")
    Lire(su)

    s <- 39 * su
    s1 <- s + ((s * 50) / 100)
    s2 <- s + ((s * 75) / 100)
    s3 <- s * 2

    Ecrire("Donner la salaire : ")
    Lire(sd)

    Si sd >= s1 et sd < s2 Alors
        Ecrire("il a travaillé entre 40 heures et 44 heures")
    FinSi

    Si sd >= s2 et sd < s3 Alors
        Ecrire("il a travaillé entre 45 heures et 49 heures")
    FinSi

    Si sd >= s3 Alors
        Ecrire("il a travaillé 50 heures ou plus")
    FinSi

    Si sd < s1 Alors
        Ecrire("Il travaille exactement !")
    FinSi
Fin

//EXERCICE3
Algorithme IncrementDate

Variables
    day: entier
    month: entier
    year: entier

Début
    Ecrire("Entrez le jour : ")
    Lire(day)
    Ecrire("Entrez le mois : ")
    Lire(month)
    Ecrire("Entrez l'année : ")
    Lire(year)

    # Incrémenter le jour
    day <- day + 1

    # Déterminer le nombre maximum de jours dans le mois courant
    max_jours <- 0

    Si month = 1 ou month = 3 ou month = 5 ou month = 7 ou month = 8 ou month = 10 ou month = 12 Alors
        max_jours <- 31
    SinonSi month = 2 Alors
        Si (year % 4 = 0 et year % 100 <> 0) ou (year % 400 = 0) Alors
            max_jours <- 29
        Sinon
            max_jours <- 28
        FinSi
    Sinon
        max_jours <- 30
    FinSi

    # Vérifier si le jour dépasse le maximum du mois courant
    Si day > max_jours Alors
        day <- 1
        month <- month + 1
    FinSi

    # Vérifier si le mois dépasse 12
    Si month > 12 Alors
        month <- 1
        year <- year + 1
    FinSi

    Ecrire("Jour : ", day, " Mois : ", month, " Année : ", year)
Fin
